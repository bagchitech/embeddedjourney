cmake_minimum_required(VERSION 3.16)
project(EmbeddedJourney C)
enable_testing()

set(CMAKE_C_STANDARD 11)

add_compile_options(
    -Wall
    -Wextra
    -g
)

# Unity as a static library — built once, linked by all tests
add_library(unity STATIC
    unity/unity.c
)
target_include_directories(unity PUBLIC .)

add_library(practice2504 STATIC
    src/practice2504.c
)
target_include_directories(practice2504 PUBLIC include/)

# Pull in the tests folder
add_subdirectory(tests)
